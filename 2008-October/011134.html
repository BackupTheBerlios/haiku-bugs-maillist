Comment(by kaliber):

 It is possible to convert comments automatically e.g on Ubuntu:


 {{{
 #!/bin/sh

 for i in `find . -name '*.h'`
 do
     fix=0
     grep "class" $i || fix=1
     if [ "$fix" = 1 ]
     then
         mv $i $i.bak
         ccmtcnvt $i.bak > $i
         rm -f $i.bak
     fi
 done

 }}}
